461  case OB_FRAME_COLOR:

463  case OB_FRAME_GYRO:
464  return (typeid(T) == typeid(GyroFrame));
465  case OB_FRAME_ACCEL:
466  return (typeid(T) == typeid(AccelFrame));
467  case OB_FRAME_SET:
