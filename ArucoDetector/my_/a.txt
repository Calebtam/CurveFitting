// main.cpp
#include <iostream>
#include <string>
 
int main(int argc, char** argv) {
    if (argc < 3) {
        std::cerr << "Usage: " << argv[0] << " path_to_file number" << std::endl;
        return 1;
    }
 
    std::string path = argv[1];
    int number = std::stoi(argv[2]);
 
    std::cout << "Path: " << path << std::endl;
    std::cout << "Number: " << number << std::endl;
 
    return 0;
}
